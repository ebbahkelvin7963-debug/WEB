import { GoogleGenAI, Type } from "@google/genai";
import { BibleTranslation, ScriptureResponse, NameMeaningResponse, DailyChapterResponse } from "../types";

const apiKey = process.env.API_KEY || '';
const ai = new GoogleGenAI({ apiKey });

const modelName = 'gemini-2.5-flash';

export const getScripture = async (reference: string, translation: BibleTranslation): Promise<ScriptureResponse> => {
  try {
    const prompt = `Provide the full text of the Bible passage ${reference} in the ${translation} translation. Also provide a very brief (1 sentence) historical context or commentary.`;
    
    const response = await ai.models.generateContent({
      model: modelName,
      contents: prompt,
      config: {
        responseMimeType: "application/json",
        responseSchema: {
          type: Type.OBJECT,
          properties: {
            reference: { type: Type.STRING },
            translation: { type: Type.STRING },
            text: { type: Type.STRING },
            commentary: { type: Type.STRING }
          },
          required: ["reference", "translation", "text"]
        }
      }
    });

    const text = response.text;
    if (!text) throw new Error("No data returned");
    return JSON.parse(text) as ScriptureResponse;
  } catch (error) {
    console.error("Error fetching scripture:", error);
    throw error;
  }
};

export const getNameMeaning = async (name: string): Promise<NameMeaningResponse> => {
  try {
    const prompt = `Explain the biblical meaning, origin, and significance of the name "${name}". List up to 3 key scriptural references where this name or a character with this name appears.`;
    
    const response = await ai.models.generateContent({
      model: modelName,
      contents: prompt,
      config: {
        responseMimeType: "application/json",
        responseSchema: {
          type: Type.OBJECT,
          properties: {
            name: { type: Type.STRING },
            meaning: { type: Type.STRING },
            scripturalReferences: { 
              type: Type.ARRAY,
              items: { type: Type.STRING }
            }
          }
        }
      }
    });

    const text = response.text;
    if (!text) throw new Error("No data returned");
    return JSON.parse(text) as NameMeaningResponse;
  } catch (error) {
    console.error("Error fetching name meaning:", error);
    throw error;
  }
};

export const getDailyChapter = async (): Promise<DailyChapterResponse> => {
  try {
    const prompt = `Select a random chapter from the Bible (Old or New Testament). Provide the Book and Chapter number, a summary of the chapter, and 2-3 key verses from it in the NIV translation.`;
    
    const response = await ai.models.generateContent({
      model: modelName,
      contents: prompt,
      config: {
        responseMimeType: "application/json",
        responseSchema: {
          type: Type.OBJECT,
          properties: {
            book: { type: Type.STRING },
            chapter: { type: Type.NUMBER },
            summary: { type: Type.STRING },
            keyVerses: { type: Type.STRING }
          }
        }
      }
    });

    const text = response.text;
    if (!text) throw new Error("No data returned");
    return JSON.parse(text) as DailyChapterResponse;
  } catch (error) {
    console.error("Error fetching daily chapter:", error);
    throw error;
  }
};
